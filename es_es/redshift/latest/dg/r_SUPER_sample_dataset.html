El esquema de la tabla y el modelo de datos utilizados para los ejemplos de ingesta y consulta se definen de la siguiente manera.

/*customer-orders-lineitem*/ 
CREATE TABLE customer_orders_lineitem
(c_custkey bigint
,c_name varchar
,c_address varchar
,c_nationkey smallint
,c_phone varchar
,c_acctbal decimal(12,2)
,c_mktsegment varchar
,c_comment varchar
,c_orders super
);

ARRAY < STRUCT < o_orderkey:bigint
